import axios from 'axios'

export function request(config){
   const instance = new axios.create({
       baseURL:
       timeout:
   }) 
   instance.interceptors.request.use(config=>{
       return config
   },err=>{
       console.log(err)
   })
   instance.interceptors.respones.use(res=>{
       return res.data
   },err=>{
       console.log(err)
   })
   return instance(config)
}
