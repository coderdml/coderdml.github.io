//router
const Home = () => import('../views/home/Home')  //懒加载
Vue.use(VueRouter)
const routes = [
    {
        path : '',
        redirect : '/home',   //默认路由
    },
    {
        path : '/home',
        component : Home
    }
const router = new VueRouter({
    routes,
    mode : "history"    //设置history取代哈希
}
)


//<slot>的使用
//在整个tabbar设置插槽供给各个item使用
<template>
    <div id="tab-bar">
        <slot></slot>
      </div>
</template>
//需要给插槽设置限定条件时使用<div/>标签包裹起来以防止插槽被整个替换
<template>
    <div class="tab-bar-item"> 
        <div v-if="!isActive"><slot  name = "item-icon"></slot></div>
        <div v-else><slot name = "item-icon-active"></slot></div>
        <div :class="{active:isActive}"><slot name = "item-text"></slot></div>
    </div>  
</template>


//css
#tab-bar {
    display: flex;
    background-color: #f6f6f6;
    position: fixed;
    left: 0;
    right: 0;
    bottom: 0;
    box-shadow: 0 -1px 2px rgba(100, 100, 100, 0.2);  //x方向，y方向，厚度，透明度
}
.tab-bar-item {
    flex : 1; 
    text-align: center;
    font-size:14px;
    height: 49px;  //通用高度为49px
}
.tab-bar-item img {
    height : 22px;
}
